<h2>Codigo que suma dos numeros </h2>

```
#include <stdio.h>

using  namespace std;

int main() {
    int a,b,c
    printf("Programa que suma dos numeros \n");
    printf("Ingresar el primer numero : \n");
    scanf("%d" ,&a);
    printf("Ingresa el segundo numero : \n");
    scanf("%d" ,&b);
    c=a+b;
    printf("\n El resultado es : %d \n" ,c);
    return 0;
}
```
